



displays the value from the vars set in the config file

#+begin_src yaml
# in config.yml
nav:
  whatever: site.com
  whatever: site2.com
#+end_src

#+begin_src html
<div class="link-wrapper-mobile">
    {% if site.nav %}
        {% for link in site.nav %}
           {{ link[1] }}
           <span style="margin-left: 5px;"></span>
        {% endfor %}
    {% endif %}
</div>


   {% if site.nav %}
    <ul class="link">
        {% for link in site.nav %}
          <li>{{ link[1] }}</li>
        {% endfor %}
     </ul>
     {% endif %}
#+end_src
